function twoSum(numbers, target) {
  let tuple = []
  for (let i = 0; i < numbers.length - 1; i++){
    for (let j = i + 1; j < numbers.length; j++){
      if (numbers[i] + numbers[j] === target){

        tuple.push(j)
      }
      else continue
    }
  }
  return tuple
}
